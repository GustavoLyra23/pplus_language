cmake_minimum_required(VERSION 3.16)
project(portugolpp_cpp)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find required packages
find_package(PkgConfig REQUIRED)
pkg_check_modules(ANTLR4 REQUIRED antlr4-runtime)

# Include directories
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/cpp_src/include)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/cpp_src/antlr_generated/src/main/antlr/org/gustavolyra)
include_directories(/usr/local/include/antlr4-runtime)

# Collect source files
file(GLOB_RECURSE SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/cpp_src/src/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/cpp_src/antlr_generated/src/main/antlr/org/gustavolyra/*.cpp
)

# Create executable
add_executable(portugolpp_cpp ${SOURCES})

# Link libraries
target_link_libraries(portugolpp_cpp
    antlr4-runtime
)

# Set output directory
set_target_properties(portugolpp_cpp PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)

# Installation
install(TARGETS portugolpp_cpp
    RUNTIME DESTINATION bin
)
